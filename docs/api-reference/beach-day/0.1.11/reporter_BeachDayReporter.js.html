function stringLogArguments(args) {
    var newArgs = [];
    for (var a = 0; a < args.length; a++) {
        var item = args[a];
        if (typeof item == "object") {
            var result = stringifyObject(item, { singleQuotes: false }).replace(/(\r\n|\n|\r|\t)/gm, "");
            newArgs.push(result);
        }
        else {
            newArgs.push(item);
        }
    }
    return newArgs.join(", ");
}
